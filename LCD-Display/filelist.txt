  File root;
  String nameList[100];
  int count = 0;
  root = SD.open("/");
  
  while (true) {
    File entry =  root.openNextFile();
    if (! entry) {
      // no more files
      break;
    }
    
    String name = entry.name();
    int len = name.length();
  
    if (name.substring(len-3, len) == "WAV"){
      //Serial.println(name);
      nameList[count] = name.substring(0, len-4);
      count ++;
      }
    entry.close();
  }
  root.close();
